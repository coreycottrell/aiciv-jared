The Meta Tags — Copy & Paste Ready
<!-- ================================================
  OG IMAGE SPLIT STRATEGY
  Add these tags to <head> — purebrain.ai homepage
  ================================================ -->

<!-- For LinkedIn (GIF animates!) and Facebook (first frame) -->
<meta property="og:image"
      content="https://purebrain.ai/wp-content/uploads/2026/02/Pure-Brain-Vid-3.gif" />
<meta property="og:image:width"  content="480" />
<meta property="og:image:height" content="270" />
<meta property="og:image:type"  content="image/gif" />

<!-- For Twitter/X (static image overrides og:image when present) -->
<meta name="twitter:card"  content="summary_large_image" />
<meta name="twitter:image" content="https://purebrain.ai/wp-content/uploads/2026/02/purebrain-homepage-og.jpg" />
